426 
427  operation_t * op = get_operation(operation); /* identify the operation */
428 
429  if (!operation || !op) {
430  ERROR("invalid operation: %s", line);
431  } else {
432  /* count the number of the operands */
433  uint8_t number_of_operands = 0;
434  number_of_operands += operand1 ? 1 : 0;
435  number_of_operands += operand2 ? 1 : 0;
436 
437  if (number_of_operands != op->operands) {
438  ERROR("wrong number of operands at '%s', expected %u, got %u",
439  operation, op->operands, number_of_operands);
440  } else {
441  switch (op->operands) {
442  /* operations with no operands */
443  case 0: {

445
